# Совместимость типов

```ts
type ActiveDeal = { status: "ok" } | { status: "at_risk" } | { status: "payment_due" };

type ClosedDeal = { status: "completed" } | { status: "stopped" };

type Deal = ActiveDeal | CloseDeal;

type Lead = { status: "contact" } | { status: "presale" };

type Entity = Deal | Lead;

// Hierarchy
// ActiveDeal <-- Deal <-- Entity

type Processor<Input extends Entity, Output extends Entity> = (input: Input) => Output;

type DealProcessor = Processor<Deal, Deal>;

const x: DealProcessor = (input: Deal) => {
  return input;
};
```
