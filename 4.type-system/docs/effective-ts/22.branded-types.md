# Брендированные типы

```ts
type User = {
  id: number;
  name: string;
  departmentId: number;
  companyId: number;
};

// Report.ts

function generateReport(id: number) {
  // ....
}
```

**Возникает проблема - нет понимания какой id передать? `departmentId` или `companyId` или `id` пользователя**

```ts
const user = {
  id: 3,
  name: "User",
  departmentId: 1,
  companyId: 2,
};

// Какой id мы должны передать ?!
generateReport(user.id);
generateReport(user.companyId);
generateReport(user.departmentId);
```

Один из вариантов решения - жестко связать тип аргумента функции с типом сущности

```ts
function generateReport(id: User["id"]) {
  // ....
}
```

```ts
// Этого никогда не делаем сами, за нас это делает библиотека

type Brand<T, Name extends string> = T & { __brand: Name };

type Demo = number & { __brand: "something" };

const t: Demo = 3; // Type 'number' is not assignable to type 'Demo'
const x: Demo = 3 as Demo; // No error

const sum = x + 1; // 4
```

Нам могут помочь **брендированные типы**

```ts
type Brand<T, Name extends string> = T & { __brand: Name };

type UserId = Brand<number, "UserId">;

type CompanyId = Brand<number, "CompanyId">;

type User = {
  id: UserId;
  name: string;
  departmentId: number;
  companyId: CompanyId;
};

// БИБЛИОТЕЧНЫЙ КОД

function brandify<BrandedType extends Brand<any, string>>(
  value: Omit<BrandedType, "__brand">,
): BrandedType {
  return value as BrandedType;
}

// КОНЕЦ БИБЛИОТЕЧНОГО КОДА

const user: User = {
  id: brandify<UserId>(3),
  name: "User",
  departmentId: 1,
  companyId: brandify<CompanyId>(2),
};

function generateReport(id: User["id"]) {
  // ....
}
```
