```ts
type User = {
  name: string;
  email: string;
};

type UserBuilderState = {
  hasName: boolean;
  hasEmail: boolean;
};

type UserBuilder<State extends UserBuilderState = { hasName: false; hasEmail: false }> = {
  setName(name: string): UserBuilder<Omit<State, "hasName"> & { hasName: true }>;
  setEmail(email: string): UserBuilder<Omit<State, "hasEmail"> & { hasEmail: true }>;
  build: State extends { hasName: true; hasEmail: true } ? () => User : "Missing required field";
};

declare let builder: UserBuilder;
const user1 = builder.setName("some").build(); // Missing required field
const user2 = builder.setName("name").setEmail("email").build(); // Correct
```

```ts
type YesOrNo<T> = T extends Promise<any> ? "yes" : no;
type Test = YesOrNo<Promise<number> | number>; // Result - `yes | no`
```

Проверка в типе `YesOrNo` будет проходить для каждого типа переданного в дженерике. Это можно пофиксить таким хаком:
Превращаем элементы в `tuple`

```ts
type YesOrNo<T> = [T] extends [Promise<any>] ? "yes" : no;
type Test = YesOrNo<Promise<number> | number>; // Result -  `no`
```
